# Function: createApprovalTypedData()

```ts
function createApprovalTypedData(params): {
  domain: {
     chainId: number;
     name: "ECP";
     verifyingContract: `0x${string}`;
     version: "1";
  };
  message: {
     app: `0x${string}`;
     author: `0x${string}`;
     deadline: bigint;
     expiry: bigint;
     nonce: bigint;
  };
  primaryType: "AddApproval";
  types: {
     AddApproval: (
        | {
        name: "author";
        type: "address";
      }
        | {
        name: "app";
        type: "address";
      }
        | {
        name: "nonce";
        type: "uint256";
      }
        | {
        name: "deadline";
        type: "uint256";
      }
        | {
        name: "expiry";
        type: "uint256";
     })[];
  };
};
```
