# Function: createReportCommentTypedData()

```ts
function createReportCommentTypedData(params): {
  domain: {
     chainId: number;
     name: "Ethereum Comments Protocol";
     verifyingContract: `0x${string}`;
     version: "1";
  };
  message: {
     commentId: `0x${string}`;
     message: string;
     reportee: `0x${string}`;
  };
  primaryType: "ReportComment";
  types: {
     ReportComment: (
        | {
        name: "commentId";
        type: "bytes32";
      }
        | {
        name: "reportee";
        type: "address";
      }
        | {
        name: "message";
        type: "string";
     })[];
  };
};
```
